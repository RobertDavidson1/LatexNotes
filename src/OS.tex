  coltitle=definitionpurple,
  boxrule=0.5pt,

  bottomrule=0.5pt,
  left=12pt,
  right=12pt,
  top=8pt,
  bottom=8pt,
  attach boxed title to top left={xshift=0mm, yshift=0mm},
  boxed title style={size=small, colback=definitionpurple},
  title={Definition},
  overlay unbroken and first={
    \node[anchor=west, font=\bfseries\sffamily, text=definitionpurple, 
          inner sep=0pt, outer sep=0pt] 
          at ([xshift=2.5cm, yshift=-8pt]frame.north west) {#2};
  },
  #1
}
